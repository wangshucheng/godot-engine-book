    # 网格
    var surfaces: Array

    # 材质
    var shader: Shader
    var shader_uniforms: Dictionary
}

    # 物理体
    var collision_layer: int
    var collision_mask: int
}

class AnimationPlayer {
    # 动画播放器
    var animations: Array
    var playing: bool

    var stream: AudioStream
    var volume: float

    var uniforms: Array
